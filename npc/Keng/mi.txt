	mes "^0033FFเลือด ^FF0000: ^000000" + getmonsterinfo(.@id,MOB_MAXHP);
	mes "^0033FFค่าประสบการณ์ Base ^FF0000: ^000000" + getmonsterinfo(.@id,MOB_BASEEXP);
	mes "^0033FFค่าประสบการณ์ Job ^FF0000: ^000000" + getmonsterinfo(.@id,MOB_JOBEXP);
	mes "^0033FFพลังโจมตี ^FF0000: ^000000" + getmonsterinfo(.@id,MOB_ATK1);
	mes "^0033FFพลังป้องกันกายภาพ^FF0000 : ^000000" + getmonsterinfo(.@id,MOB_DEF);

	mes "^0033FFDex ^FF0000: ^000000"+ getmonsterinfo(.@id,MOB_DEX);
	mes "^0033FFLuk^FF0000 : ^000000"+ getmonsterinfo(.@id,MOB_LUK);
	mes "^0033FFระยะโจมตี ^FF0000: ^000000"+ getmonsterinfo(.@id,MOB_RANGE);

	mes "^0033FFประเภท^FF0000 : ^000000"+ .Race$[getmonsterinfo(.@id,MOB_RACE)];
	mes "^0033FFธาตุ^FF0000 : ^000000"+ .Element$[getmonsterinfo(.@id,MOB_ELEMENT)];

	{
		.@count = $@MobDrop_count;
		copyarray .@item[0],$@MobDrop_item[0],.@count;
		copyarray .@rate[0],$@MobDrop_rate[0],.@count;
	
		for( .@i = 0; .@i < .@count; .@i++ )
		{
			mes "" + getitemname(.@item[.@i]) + " " + .@rate[.@i]/100 + ((.@rate[.@i]%100 < 10) ? ".0":".") + .@rate[.@i]%100 + " %";
		}
	}
	mes "^FF0000=========================^000000";
	
	end;

	bindatcmd "mi", strnpcinfo(3) +"::OnDetail", 0, 40;

-	script	Detailpvp	FAKE_NPC,{

OnDetail:

	mes "^FF0000===== มีเวลาให้จัดการ 30 วินาที =====^000000";
	mes "^FF0000 หากทำรายการต่อคุณจะเสีย 10 Cashpoint^000000";

	if( #CASHPOINTS > 0 )
	if( select( "ใช่","ยกเลิก" ) == 1 ){
		#CASHPOINTS = #CASHPOINTS - 10;
		dispbottom "Cash Point ขณะนี้คุณมี "+#CASHPOINTS+"Cash point";
	}

	atcommand "@pvpon";
		attachnpctimer strcharinfo(0);
		startnpctimer;
	end;
OnTimer1000:
	dispbottom "After a second...";
	message strcharinfo(0),"You have "+((getnpctimer(0))/1000)+" Seconds.";
	end;

OnTimer20000:
	dispbottom "last 10 seconds...";
	end;

OnTimer30000:
	mes "^FF0000===== หมดเวลา =====^000000";
	stopnpctimer;
	atcommand "@pvpoff";
	atcommand "@revive";
	
	end;	
OnInit:
	bindatcmd "pvpon", strnpcinfo(3) +"::OnDetail", 0, 40;

	end;
}
