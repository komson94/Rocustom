1@rasan_n,155,266,3	script	Secret Merchant#rasan_n	4_M_HUMERCHANT,{
	disable_items;
	mes .NPCname$;
	mes "ข้าสามารถทำการรีเซ็ต";
	mes "^FF0000Treasure Box^000000";
	mes "ให้กับเจ้าได้ โดยค่าคิดค่าธรรมเนียม";
	mes "ทั้งหมด ^006600"+callfunc("F_InsertComma",.Price)+"^000000 z.";
	mes "เจ้าสนใจหรือไม่?";
	next;
	switch(select( ""+.sb$+" ข้าไม่สนใจ",""+.sb$+" ข้าต้องการรีเซ็ต" )) {
		case 1: end;
		case 2:
			if (#SanctuaryTreasureReset == 1) {
				mes .NPCname$;
				mes "ข้าสามารถทำการรีเซ็ต";
				mes "^FF0000Treasure Box^000000";
				mes "ได้เพียงแค่ครั้งเดียวเท่านั้น.";
				close;
			}
			if (Zeny < .Price) {
				mes .NPCname$;
				mes "เจ้ามีเงินไม่พอที่จะจ่ายค่าธรรมเนียม.";
				close;
			}
			mes .NPCname$;
			mes "ข้าได้ทำการรีเซ็ต";
			mes "^FF0000Treasure Box^000000";
			mes "ให้เจ้าเป็นที่เรียบร้อยแล้ว";
			Zeny -= .Price;
			#SanctuaryTreasure = 0;
			#SanctuaryTreasureReset = 1;
			specialeffect2 EF_MAGNIFICAT;
			end;
	}
	
OnInit:
	.NPCname$ = "[^0000FF "+strnpcinfo(1)+" ^000000]";
	.sb$ = "~";
	.Price = 100000;
	end;
}
